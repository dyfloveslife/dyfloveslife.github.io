public class Solution {
    public int digitsInSequence(int arr[], int n) {
        if (n < 0) return -1;
        if (n <= 9) return n;

        int m = 0;
        for (int i = 1; i <= 9; i++) {
            if (arr[i] >= n) {
                m = i;
                break;
            }
        }

        n = n - arr[m - 1];
        int temp1 = n / m;
        int temp2 = n % m;
        int temp3 = (int) (temp1 + Math.pow(10, m - 1));
        return (int) (temp3 / Math.pow(10, m - temp2 - 1)) % 10;
    }
